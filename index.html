<script>

  const supabaseUrl = "https://goldtrack-ebnuveu9.manus.space";

  const supabaseKey = "https://glklaxiwbhynwvvanzfk.supabase.co";

  const supabase = window.supabase.createClient(supabaseUrl, supabaseKey);


  function showAdmin(){

    document.getElementById("trackingSection").classList.add("hidden");

    document.getElementById("adminSection").classList.remove("hidden");

  }


  async function trackShipment(){

    const id = document.getElementById("trackingInput").value;

    const { data } = await supabase.from("shipments").select("*").eq("tracking_id", id).single();

    if(!data){

      document.getElementById("result").innerHTML = "Tracking not found.";

      return;

    }

    document.getElementById("result").innerHTML = `

      <p><strong>Customer:</strong> ${data.customer_name}</p>

      <p><strong>Route:</strong> ${data.origin} → ${data.destination}</p>

      <p>Status: <span class="status ${data.status}">${data.status}</span></p>

      <p>Last Updated: ${new Date(data.updated_at).toLocaleString()}</p>

    `;

  }


  async function login(){

    const email = document.getElementById("email").value;

    const password = document.getElementById("password").value;

    const { error } = await supabase.auth.signInWithPassword({ email, password });

    if(error){ alert("Login failed"); return; }

    document.getElementById("adminDashboard").classList.remove("hidden");

    loadShipments();

  }


  async function logout(){

    await supabase.auth.signOut();

    location.reload();

  }


  function generateTracking(){

    return "GT-" + Math.floor(10000 + Math.random()*90000);

  }


  async function createShipment(){

    const tracking = generateTracking();

    await supabase.from("shipments").insert([{

      tracking_id: tracking,

      customer_name: document.getElementById("customer").value,

      origin: document.getElementById("origin").value,

      destination: document.getElementById("destination").value,

      status: "Ongoing",

      updated_at: new Date()

    }]);

    alert("Shipment Created: " + tracking);

    loadShipments();

  }


  async function loadShipments(){

    const { data } = await supabase.from("shipments").select("*");

    let html="";

    data.forEach(s=>{

      html += `

      <div style="border:1px solid #333;padding:10px;margin:10px 0;">

        <strong>${s.tracking_id}</strong> - ${s.customer_name}<br>

        ${s.origin} → ${s.destination}<br>

        <select onchange="updateStatus('${s.tracking_id}', this.value)">

          <option ${s.status=="Ongoing"?"selected":""}>Ongoing</option>

          <option ${s.status=="Held"?"selected":""}>Held</option>

          <option ${s.status=="Stopped"?"selected":""}>Stopped</option>

          <option ${s.status=="Delivered"?"selected":""}>Delivered</option>

        </select>

      </div>

      `;

    });

    document.getElementById("shipmentsList").innerHTML=html;

  }


  async function updateStatus(id,status){

    await supabase.from("shipments").update({status:status,updated_at:new Date()}).eq("tracking_id",id);

    loadShipments();

  }
